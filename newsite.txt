// ── CANVAS — VIOLET AMBIENT ──────────────────────────────────────────────
const cv = document.getElementById('canvas');
const cc = cv.getContext('2d');
let W, H, particles = [];

function resize() { W=cv.width=window.innerWidth; H=cv.height=window.innerHeight; }
resize(); window.addEventListener('resize', ()=>{ resize(); spawnParticles(); });

function spawnParticles() {
  particles = [];
  const n = Math.floor(W*H/16000);
  for(let i=0;i<n;i++) particles.push({
    x: Math.random()*W, y: Math.random()*H,
    vx: (Math.random()-.5)*.18, vy: (Math.random()-.5)*.18,
    r: Math.random()*.9+.2,
    a: Math.random()*.3+.05,
    phase: Math.random()*Math.PI*2,
    speed: .006+Math.random()*.012
  });
}
spawnParticles();

// violet aurora orbs
const orbs = [
  { ox:.18, oy:.35, dx:.06, dy:.04, t:0, ts:.005, r:.45, a:.05 },
  { ox:.78, oy:.62, dx:.05, dy:.07, t:2.1, ts:.006, r:.38, a:.035 },
  { ox:.5,  oy:.1,  dx:.04, dy:.05, t:3.8, ts:.004, r:.3,  a:.025 },
  { ox:.88, oy:.15, dx:.07, dy:.04, t:1.2, ts:.007, r:.22, a:.02 },
];

function drawParticleConnections() {
  const d = 110;
  for(let i=0;i<particles.length;i++) {
    for(let j=i+1;j<particles.length;j++) {
      const dx=particles[i].x-particles[j].x, dy=particles[i].y-particles[j].y;
      const dist=Math.sqrt(dx*dx+dy*dy);
      if(dist<d) {
        cc.strokeStyle=`rgba(124,58,237,${(1-dist/d)*.04})`;
        cc.lineWidth=.5;
        cc.beginPath(); cc.moveTo(particles[i].x,particles[i].y);
        cc.lineTo(particles[j].x,particles[j].y); cc.stroke();
      }
    }
  }
}

let T=0;
(function frame() {
  T+=.005;
  cc.clearRect(0,0,W,H);

  // orbs
  orbs.forEach(o => {
    o.t += o.ts;
    const bx=(o.ox+Math.sin(o.t)*o.dx)*W;
    const by=(o.oy+Math.cos(o.t*1.3)*o.dy)*H;
    const g=cc.createRadialGradient(bx,by,0,bx,by,Math.min(W,H)*o.r);
    g.addColorStop(0, `rgba(124,58,237,${o.a})`);
    g.addColorStop(.5, `rgba(91,33,182,${o.a*.4})`);
    g.addColorStop(1, 'transparent');
    cc.fillStyle=g; cc.fillRect(0,0,W,H);
  });

  drawParticleConnections();

  // particles
  particles.forEach(p => {
    p.x+=p.vx; p.y+=p.vy; p.phase+=p.speed;
    if(p.x<0||p.x>W) p.vx*=-1;
    if(p.y<0||p.y>H) p.vy*=-1;
    const a = p.a*(.6+.4*Math.sin(p.phase));
    const r = p.r*(.88+.12*Math.sin(p.phase*1.5));
    cc.beginPath(); cc.arc(p.x,p.y,r,0,Math.PI*2);
    cc.fillStyle=`rgba(157,95,252,${a})`; cc.fill();
  });
  requestAnimationFrame(frame);
})();

const msgsEl = document.getElementById('chatMsgs');
let chatRunning = false;

function makeMsg(m) {
  const w = document.createElement('div');
  w.className = `msg ${m.role}`;
  w.innerHTML = `<div class="msg-label">${m.role==='user'?'You':'targetdash AI'}</div>
    <div class="bubble">${m.text}</div>`;
  return w;
}

function runChat() {
  if(chatRunning) return; chatRunning = true;
  msgsEl.innerHTML = '';
  let delay = 500;

  CONVO.forEach((m, i) => {
    setTimeout(() => {
      if(m.role === 'ai') {
        const typing = document.createElement('div');
        typing.className = 'msg ai visible';
        typing.innerHTML = `<div class="msg-label">targetdash AI</div>
          <div class="typing-bubble"><span></span><span></span><span></span></div>`;
        msgsEl.appendChild(typing);
        msgsEl.scrollTop = msgsEl.scrollHeight;
        setTimeout(() => {
          typing.remove();
          const el = makeMsg(m);
          msgsEl.appendChild(el);
          requestAnimationFrame(() => el.classList.add('visible'));
          msgsEl.scrollTop = msgsEl.scrollHeight;
        }, 1000);
      } else {
        const el = makeMsg(m);
        msgsEl.appendChild(el);
        requestAnimationFrame(() => el.classList.add('visible'));
        msgsEl.scrollTop = msgsEl.scrollHeight;
      }
    }, delay);
    delay += m.role === 'ai' ? 2800 : 1600;
  });

  setTimeout(() => { chatRunning = false; runChat(); }, delay + 4000);
}

const obs = new IntersectionObserver(entries => {
  entries.forEach(e => {
    if(e.isIntersecting) {
      e.target.classList.add('in');
      if(e.target.id === 'chatBox') setTimeout(runChat, 800);
      obs.unobserve(e.target);
    }
  });
}, { threshold: .1 });
